[docs]    def data(self, index, role):
        """Return the data for the given index and role."""
        if role == Qt.DisplayRole:
            value = self._data.iloc[index.row(), index.column()]
            return str(value)

[docs]    def flags(self, index):
        """Return the flags for the given index."""
        if not index.isValid():
            return Qt.ItemIsEnabled
        return (
            super().flags(index) | Qt.ItemIsEditable | Qt.ItemIsSelectable
        )  # add editable and selectable flag.



[docs]    def headerData(self, section, orientation, role):
        """Return the header data for the given section."""
        # section is the index of the column/row.
        if role == Qt.DisplayRole:
            if orientation == Qt.Horizontal:
                return str(self._data.columns[section])

            if orientation == Qt.Vertical:
                return str(self._data.index[section])



[docs]    def setData(self, index, value, role):
        """Set the data for the given index and role."""
        if role == Qt.EditRole:
            # Set the value into the frame.
            self._data.iloc[index.row(), index.column()] = value
            self.dataChanged.emit(index, index, [role])
            return True
        return False
